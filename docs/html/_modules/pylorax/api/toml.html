[docs]def loads(s):
    if isinstance(s, bytes):
        s = s.decode('utf-8')
    try:
        return toml.loads(s)
    except toml.TomlDecodeError as e:
        raise TomlError(e.msg, e.doc, e.pos)



[docs]def dumps(o):
    # strip the result, because `toml.dumps` adds a lot of newlines
    return toml.dumps(o, encoder=toml.TomlEncoder(dict)).strip()
